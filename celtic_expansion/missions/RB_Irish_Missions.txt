	# trigger = all of ulster not owned by england, GB, scotland or their vassals is owned by you
	# effect = ire_unified_ulster_clans, army tradition, manpower, claims 
	ire_conquer_ulster = {
		icon = mission_rb_conquer_ulster
		position = 2
		required_missions = { ire_strengthen_our_army }
		provinces_to_highlight = {
			area = ulster_area
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			ulster_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			add_country_modifier = {
				name = "ire_unified_ulster_clans"
				duration = 9125 #25 years
			}
			add_army_tradition = 20
			add_manpower = 2
			connacht_area = {
				if = {
					limit = {
						NOT = { is_permanent_claim = ROOT }
						NOT = { is_core = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
			leinster_area = {
				if = {
					limit = {
						NOT = { is_permanent_claim = ROOT }
						NOT = { is_core = ROOT }
						NOT = {
							owner = {
								capital_scope = {
									area = ulster_area
									area = connacht_area
									area = leinster_area
									area = munster_area
								}
							}
						}
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}

	# position = 6
	# highlight = none
	# required = ire_an_irish_fleet
	# trigger = flagship or 8 heavies
	# effect = navy tradition, prestige
	ire_launch_a_flagship = {
		icon = mission_launch_a_flagship
		position = 6
